    ```javascript
      if (_monacoEditor) {
        for (var s = 0; s < _ideTabs.length; s++) {
          if (_ideTabs[s].model && _monacoEditor.getModel() === _ideTabs[s].model) {
            _ideTabs[s].viewState = _monacoEditor.saveViewState();
            break;
          }
        }
      }
      // ... rest of existing function
    ```
