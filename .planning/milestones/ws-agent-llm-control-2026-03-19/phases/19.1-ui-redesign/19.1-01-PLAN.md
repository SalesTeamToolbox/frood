try:
    while True:
        msg = await websocket.receive_text()
        # Check for JSON resize message before forwarding to stdin
        try:
            parsed = _json.loads(msg)
            if isinstance(parsed, dict) and parsed.get("type") == "resize":
                cols = int(parsed.get("cols", 80))
                rows = int(parsed.get("rows", 24))
                # Best-effort: send ANSI resize escape to process stdin
                if proc.stdin and not proc.stdin.is_closing():
                    proc.stdin.write(f"\x1b[8;{rows};{cols}t".encode())
                    await proc.stdin.drain()
                continue  # Do not forward resize message as terminal input
        except (_json.JSONDecodeError, ValueError, TypeError):
            pass  # Not JSON — fall through to normal input
        if proc.stdin and not proc.stdin.is_closing():
            proc.stdin.write(msg.encode("utf-8"))
            await proc.stdin.drain()
except Exception:
    pass
```

"""Tests for /api/remote/status endpoint."""
import pytest
from unittest.mock import patch


class TestRemoteStatus:
    """Test the /api/remote/status API endpoint."""

    @pytest.mark.asyncio
    async def test_remote_available_when_host_set(self, test_client):
        """Returns available=True when AGENT42_REMOTE_HOST is set."""
        with patch.dict("os.environ", {"AGENT42_REMOTE_HOST": "myserver.example.com"}):
            resp = test_client.get(
                "/api/remote/status",
                headers={"Authorization": "Bearer test-token"},
            )
        assert resp.status_code == 200
        data = resp.json()
        assert data["available"] is True
        assert data["host"] == "myserver.example.com"

    @pytest.mark.asyncio
    async def test_remote_unavailable_when_host_empty(self, test_client):
        """Returns available=False when AGENT42_REMOTE_HOST is empty string."""
        with patch.dict("os.environ", {"AGENT42_REMOTE_HOST": ""}, clear=False):
            resp = test_client.get(
                "/api/remote/status",
                headers={"Authorization": "Bearer test-token"},
            )
        assert resp.status_code == 200
        data = resp.json()
        assert data["available"] is False
        assert data["host"] is None

    @pytest.mark.asyncio
    async def test_remote_unavailable_when_host_not_set(self, test_client):
        """Returns available=False when AGENT42_REMOTE_HOST is not in env."""
        env = {k: v for k, v in __import__("os").environ.items() if k != "AGENT42_REMOTE_HOST"}
        with patch.dict("os.environ", env, clear=True):
            resp = test_client.get(
                "/api/remote/status",
                headers={"Authorization": "Bearer test-token"},
            )
        assert resp.status_code == 200
        data = resp.json()
        assert data["available"] is False
```
