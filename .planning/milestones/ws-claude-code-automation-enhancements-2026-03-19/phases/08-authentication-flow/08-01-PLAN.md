**Changes:**
```python
from jose.exceptions import ExpiredSignatureError, JOSEError

def _validate_jwt(token: str) -> AuthContext:
    try:
        payload = jwt.decode(token, settings.jwt_secret, algorithms=[ALGORITHM])
        username = payload.get("sub")
        if username is None:
            raise HTTPException(
                status_code=401,
                detail={"code": "missing_subject", "message": "Invalid token format"}
            )
        return AuthContext(user=username, auth_type="jwt")
    except ExpiredSignatureError:
        raise HTTPException(
            status_code=401,
            detail={"code": "token_expired", "message": "Session expired. Please log in again."}
        )
    except JOSEError:
        raise HTTPException(
            status_code=401,
            detail={"code": "invalid_token", "message": "Invalid authentication. Please log in."}
        )
```

**In `dashboard/server.py` login() function:**

Change return from direct dict to Response with cookie:
```python
from fastapi import Response

@app.post("/api/login")
async def login(req: LoginRequest, request: Request, response: Response):
    # ... existing validation ...

    token = create_token(settings.dashboard_username)

    # Set httpOnly cookie
    response.set_cookie(
        key="access_token",
        value=token,
        httponly=True,
        secure=request.url.scheme == "https",
        samesite="lax",
        max_age=86400  # 24 hours
    )

    return {
        "status": "ok",
        "token": token,  # Keep for non-browser clients
        "message": "Login successful"
    }
```
