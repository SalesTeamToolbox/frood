    **Add module-level dict and helper functions to core/task_context.py** (after the `get_task_context()` function, at the bottom of the file):

    ```python
    # Phase 43: Per-task tool accumulation for pattern detection
    # Plain dict (not ContextVar) keyed by task_id — safe because each task has
    # a unique UUID task_id, and both writers (ToolRegistry.execute) and readers
    # (end_task flush) access this module-level dict within the same process.
    _current_task_tools: dict[str, list[str]] = {}


    def append_tool_to_task(task_id: str, tool_name: str) -> None:
        """Add a tool name to the accumulator for the given task_id.

        Called by ToolRegistry.execute() after each tool invocation.
        Non-mutating when task_id is empty or None.
        """
        if task_id:
            if task_id not in _current_task_tools:
                _current_task_tools[task_id] = []
            _current_task_tools[task_id].append(tool_name)


    def pop_task_tools(task_id: str) -> list[str]:
        """Remove and return the tool list for the given task_id.

        Called by end_task() to flush accumulated tools. Always pops the key
        to prevent memory leaks even if the downstream flush fails.
        Returns empty list if task_id not found.
        """
        return _current_task_tools.pop(task_id, [])
    ```

    # --- Task-context accumulator tests ---

    def test_append_tool_to_task():
        from core.task_context import append_tool_to_task, pop_task_tools, _current_task_tools
        task_id = "test-accumulator-001"
        # Clean state
        _current_task_tools.pop(task_id, None)

        append_tool_to_task(task_id, "http_client")
        append_tool_to_task(task_id, "data_tool")
        assert _current_task_tools[task_id] == ["http_client", "data_tool"]
