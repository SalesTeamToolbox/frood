    ```python
    def end_task(ctx: TaskContext) -> None:
        """End a task context. Clears task_id and task_type from the ContextVar.

        Phase 43: Also flushes accumulated tool names to EffectivenessStore
        for pattern detection. Uses fire-and-forget asyncio.create_task() to
        never block the caller. Always pops the accumulator entry, even if
        flushing fails, to prevent memory leaks.
        """
        # Phase 43: Pop tool accumulator FIRST (always, even if flush fails)
        tool_names = pop_task_tools(ctx.task_id)

        # Phase 43: Fire-and-forget sequence flush
        if tool_names and len(tool_names) >= 2:
            try:
                import asyncio
                from memory.effectiveness import get_shared_store
                store = get_shared_store()
                if store:
                    asyncio.create_task(store.record_sequence(
                        agent_id="",  # Agent ID not available in TaskContext — grouped by task_type only
                        task_type=ctx.task_type.value if ctx.task_type else "general",
                        tool_names=tool_names,
                    ))
            except Exception:
                pass  # Non-critical — never block task cleanup for pattern tracking

        _task_id_var.reset(ctx._id_token)
        _task_type_var.reset(ctx._type_token)
        _remove_task_file()
    ```

    Note on agent_id: TaskContext does not have an agent_id field. Adding one would require updating all begin_task() call sites across the codebase — a scope expansion beyond Phase 43. Instead, pass `agent_id=""` which groups patterns by task_type only. This is acceptable because:

    - With agent_id="" all patterns are global, which is correct for pattern detection (the same tool chain is worth offloading regardless of which agent runs it)
