```typescript
/**
 * types.ts -- TypeScript interfaces mirroring Agent42 sidecar Pydantic models.
 *
 * CRITICAL: top_k and score_threshold are snake_case (no camelCase alias on Python side).
 * All other fields use camelCase per sidecar Pydantic aliases.
 */

// -- Health --
export interface SidecarHealthResponse {
  status: string;
  memory: { available: boolean; [key: string]: unknown };
  providers: { available: boolean; [key: string]: unknown };
  qdrant: { available: boolean; [key: string]: unknown };
}

// -- Memory Recall (POST /memory/recall) --
export interface MemoryRecallRequest {
  query: string;
  agentId: string;
  companyId?: string;
  top_k?: number;           // snake_case -- NO alias on Python side (Pitfall 3)
  score_threshold?: number;  // snake_case -- NO alias on Python side (Pitfall 3)
}

export interface MemoryItem {
  text: string;
  score: number;
  source: string;
  metadata: Record<string, unknown>;
}

export interface MemoryRecallResponse {
  memories: MemoryItem[];
}

// -- Memory Store (POST /memory/store) --
export interface MemoryStoreRequest {
  text: string;
  section?: string;
  tags?: string[];
  agentId: string;
  companyId?: string;
}

export interface MemoryStoreResponse {
  stored: boolean;
  point_id: string;  // snake_case -- NO alias on Python side
}

// -- Routing Resolve (POST /routing/resolve) --
export interface RoutingResolveRequest {
  taskType: string;
  agentId: string;
  qualityTarget?: string;
}

export interface RoutingResolveResponse {
  provider: string;
  model: string;
  tier: string;
  taskCategory: string;
}

// -- Effectiveness (POST /effectiveness/recommendations) --
export interface EffectivenessRequest {
  taskType: string;
  agentId?: string;
}

export interface ToolEffectivenessItem {
  name: string;
  successRate: number;
  observations: number;
}

export interface EffectivenessResponse {
  tools: ToolEffectivenessItem[];
}

// -- MCP Tool Proxy (POST /mcp/tool) --
export interface MCPToolRequest {
  toolName: string;
  params: Record<string, unknown>;
}

export interface MCPToolResponse {
  result: unknown;
  error: string | null;
}
```

**Create `plugins/agent42-paperclip/src/client.ts`** (per D-02, research Pattern 5):

Plugin's own Agent42Client, following the adapter client pattern. Constructor takes `baseUrl`, `bearerToken`, and `timeoutMs`. All methods use native fetch + AbortController. Retry once on 5xx for non-execute endpoints (matching adapter D-15 pattern).

Methods:
- `health(): Promise<SidecarHealthResponse>` -- GET /sidecar/health, no auth, 5s timeout, 1 retry on 5xx
- `memoryRecall(body: MemoryRecallRequest): Promise<MemoryRecallResponse>` -- POST /memory/recall, Bearer auth, 10s timeout, 1 retry
- `memoryStore(body: MemoryStoreRequest): Promise<MemoryStoreResponse>` -- POST /memory/store, Bearer auth, 10s timeout, 1 retry
- `routeTask(body: RoutingResolveRequest): Promise<RoutingResolveResponse>` -- POST /routing/resolve, Bearer auth, 10s timeout, 1 retry
- `toolEffectiveness(body: EffectivenessRequest): Promise<EffectivenessResponse>` -- POST /effectiveness/recommendations, Bearer auth, 10s timeout, 1 retry
- `mcpTool(body: MCPToolRequest): Promise<MCPToolResponse>` -- POST /mcp/tool, Bearer auth, 10s timeout, 1 retry
- `destroy(): void` -- no-op (for clean shutdown; no persistent connections with native fetch)

Private helpers (copy from adapter pattern):
- `authHeaders(): Record<string, string>` -- Content-Type + Authorization: Bearer
- `fetchWithTimeout(url, init, timeoutMs): Promise<Response>` -- AbortController wrapper
- `fetchWithRetry(url, init, timeoutMs): Promise<Response>` -- retry once on 5xx after 1s

Use the `timeoutMs` constructor parameter as the default timeout for POST endpoints. Health always uses 5s. All POST endpoints use `this.timeoutMs` (defaults to 10000).
