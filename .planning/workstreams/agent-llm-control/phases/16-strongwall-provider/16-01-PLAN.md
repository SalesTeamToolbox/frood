```python
class ProviderType(str, Enum):
    # ... existing entries ...
    SAMBANOVA = "sambanova"
    TOGETHER = "together"
    CUSTOM = "custom"
    # ADD: STRONGWALL = "strongwall"

class ModelTier(str, Enum):
    FREE = "free"
    CHEAP = "cheap"
    PREMIUM = "premium"

@dataclass(frozen=True)
class ProviderSpec:
    provider_type: ProviderType
    base_url: str
    api_key_env: str
    display_name: str
    default_model: str = ""
    requires_model_prefix: bool = False
    supports_function_calling: bool = True

@dataclass(frozen=True)
class ModelSpec:
    model_id: str
    provider: ProviderType
    max_tokens: int = 4096
    temperature: float = 0.3
    display_name: str = ""
    tier: ModelTier = ModelTier.FREE
    max_context_tokens: int = 128000
```
