    **Queue progress callback (D-19)** — Create a module-level callback holder and helper:
    ```python
    import logging
    logger = logging.getLogger(__name__)

    # Module-level holder for queue progress callback.
    # The caller (e.g., design router or frontend SSE endpoint) can set this
    # before calling generate_image() to receive queue status updates.
    _queue_progress_callback = None

    def set_queue_progress_callback(callback):
        """Set a callback function that receives queue status updates.
        Callback signature: callback(status: str) where status is one of
        'Queued', 'InProgress', 'Completed'.
        """
        global _queue_progress_callback
        _queue_progress_callback = callback

    def clear_queue_progress_callback():
        """Clear the queue progress callback after generation completes."""
        global _queue_progress_callback
        _queue_progress_callback = None

    def _on_queue_update(update):
        """Internal callback passed to fal_client.subscribe(on_queue_update=...).
        Translates fal.ai queue update objects into simple status strings
        and forwards to the registered callback.

        fal.ai queue update has a .status attribute: 'IN_QUEUE', 'IN_PROGRESS', 'COMPLETED'.
        """
        if _queue_progress_callback is None:
            return
        try:
            status = getattr(update, 'status', str(update))
            # Normalize to user-friendly labels
            if status == 'IN_QUEUE':
                _queue_progress_callback('Queued')
            elif status == 'IN_PROGRESS':
                _queue_progress_callback('InProgress')
            elif status == 'COMPLETED':
                _queue_progress_callback('Completed')
            else:
                logger.debug("Unknown queue status: %s", status)
        except Exception as e:
            logger.debug("Queue progress callback error: %s", e)
    ```

    **Local file download (D-13)** — New function `download_and_store(url: str, user_id: int, design_id: str, suffix: str = ".png") -> str`:
    ```python
    import aiofiles
    from pathlib import Path

    async def download_and_store(url: str, user_id: int, design_id: str, suffix: str = ".png") -> str:
        """Download an image from URL and store locally.
        Returns the relative path from the app root (for serving via StaticFiles).
        fal.ai URLs expire after ~7 days, so download immediately.
        """
        designs_dir = Path(settings.designs_dir)
        user_dir = designs_dir / str(user_id)
        user_dir.mkdir(parents=True, exist_ok=True)

        filename = f"{design_id}{suffix}"
        filepath = user_dir / filename

        async with httpx.AsyncClient(timeout=60.0) as client:
            resp = await client.get(url)
            resp.raise_for_status()

        async with aiofiles.open(str(filepath), "wb") as f:
            await f.write(resp.content)

        return str(filepath)
    ```

    ```python
    async def process_design(image_url: str, user_id: int, design_id: str) -> dict:
        """Run the full post-generation pipeline:
        1. Remove background (BiRefNet)
        2. Upscale to print-ready (ESRGAN 4x)
        3. Download and store locally

        Returns {"local_path": str, "bg_removed_url": str, "upscaled_url": str, "total_cost_usd": float}
        """
        total_cost = 0.0

        # Step 1: Background removal
        bg_result = await remove_background(image_url)
        total_cost += bg_result["cost_usd"]

        # Step 2: Upscale
        up_result = await upscale_image(bg_result["url"], scale=4)
        total_cost += up_result["cost_usd"]

        # Step 3: Download to local storage
        local_path = await download_and_store(up_result["url"], user_id, design_id)

        return {
            "local_path": local_path,
            "bg_removed_url": bg_result["url"],
            "upscaled_url": up_result["url"],
            "width": up_result.get("width", 0),
            "height": up_result.get("height", 0),
            "total_cost_usd": total_cost,
        }
    ```
