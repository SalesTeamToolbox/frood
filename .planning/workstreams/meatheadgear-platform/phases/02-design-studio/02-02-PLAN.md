<objective>
Build the Fabric.js canvas editor for design placement, add own-image upload functionality with backend persistence, and show generation progress UI. The canvas shows a flat product template where users position their design (AI-generated or uploaded), with resize/rotate/move handles. Canvas exports PNG for the mockup step. During AI generation, the chat area shows live progress status (Queued -> Generating -> Processing).

Purpose: DES-05, DES-06, and DES-08 require a visual canvas editor, upload alternative, and D-19 requires generation progress display. This is primarily frontend work that can run parallel to the backend fal.ai rewrite.
Output: Working Fabric.js canvas in the design studio section with upload support, backend upload wiring, and generation progress.
</objective>

    function showGenerationProgress(statusText) {
      let progressEl = document.getElementById('generation-progress-msg');
      if (!progressEl) {
        progressEl = document.createElement('div');
        progressEl.id = 'generation-progress-msg';
        progressEl.className = 'chat-msg system';
        const chatMessages = document.getElementById('chat-messages');
        chatMessages.appendChild(progressEl);
        chatMessages.scrollTop = chatMessages.scrollHeight;
      }
      // Clear previous content safely
      progressEl.textContent = '';
      const wrapper = document.createElement('div');
      wrapper.className = 'generation-progress';
      const dot = document.createElement('span');
      dot.className = 'progress-dot';
      wrapper.appendChild(dot);
      wrapper.appendChild(document.createTextNode(statusText));
      progressEl.appendChild(wrapper);
    }
