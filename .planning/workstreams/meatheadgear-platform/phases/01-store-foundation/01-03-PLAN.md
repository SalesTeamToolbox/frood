    ```python
    import math

    def calculate_retail_price(printful_cost: float, target_margin: float = 0.35) -> float:
        """
        Calculate retail price to achieve target gross margin.

        Formula: retail = cost / (1 - margin)
        Then round UP to nearest $X.99

        Per CONTEXT.md decision: "Retail price = Printful base cost / (1 - 0.35)"
        "Round up to nearest $0.99 (e.g., $18.46 -> $18.99)"

        Margin verification: (retail - cost) / retail >= 0.30
        (30% is the floor; 35% is the target per CONTEXT.md)
        """
        if printful_cost <= 0:
            return 0.0

        raw_price = printful_cost / (1 - target_margin)
        # Round UP to nearest $X.99
        whole = math.ceil(raw_price)
        retail = whole - 0.01  # e.g., 21 -> 20.99
        # If rounding down to .99 drops below raw, go up one dollar
        if retail < raw_price:
            retail = whole + 0.99
        return round(retail, 2)
    ```

    **services/printful.py** — Async Printful API v2 client:
    ```python
    import httpx
    import logging
    from typing import Any

    logger = logging.getLogger("meatheadgear.printful")

    PRINTFUL_BASE_URL = "https://api.printful.com/v2"

    # Printful catalog product IDs for gym wear categories
    # These are Printful's category groupings for: t-shirts, hoodies, leggings, shorts, hats, bags
    GYM_WEAR_CATEGORY_IDS = [
        "T-shirts",
        "Hoodies & Sweatshirts",
        "Leggings",
        "Shorts",
        "Hats",
        "Bags",
    ]

    class PrintfulClient:
        def __init__(self, api_key: str):
            self._api_key = api_key
            self._headers = {
                "Authorization": f"Bearer {api_key}",
                "Content-Type": "application/json",
            }

        async def get_catalog_products(self) -> list[dict[str, Any]]:
            """Fetch all catalog products from Printful, filtered to gym wear categories."""
            products = []
            try:
                async with httpx.AsyncClient(timeout=30.0) as client:
                    resp = await client.get(
                        f"{PRINTFUL_BASE_URL}/catalog-products",
                        headers=self._headers,
                    )
                    if resp.status_code == 429:
                        logger.warning("Printful rate limited — skipping sync")
                        return []
                    resp.raise_for_status()
                    data = resp.json()
                    # Filter to gym wear categories
                    for product in data.get("data", []):
                        category = product.get("type_name", "")
                        if any(cat.lower() in category.lower() for cat in GYM_WEAR_CATEGORY_IDS):
                            products.append(product)
            except httpx.HTTPError as e:
                logger.error("Printful API error: %s", e)
            return products

        async def get_product_details(self, product_id: int) -> dict[str, Any] | None:
            """Fetch detailed product info including variants and pricing."""
            try:
                async with httpx.AsyncClient(timeout=30.0) as client:
                    resp = await client.get(
                        f"{PRINTFUL_BASE_URL}/catalog-products/{product_id}",
                        headers=self._headers,
                    )
                    if resp.status_code == 429:
                        logger.warning("Printful rate limited on product %d", product_id)
                        return None
                    resp.raise_for_status()
                    return resp.json().get("data")
            except httpx.HTTPError as e:
                logger.error("Printful product %d error: %s", product_id, e)
                return None
    ```

    **services/catalog.py** — Catalog sync orchestrator:
    ```python
    import asyncio
    import logging
    from datetime import datetime, timezone
    from services.printful import PrintfulClient
    from services.pricing import calculate_retail_price
    from config import settings
    from database import get_db, DB_PATH
    import aiosqlite

    logger = logging.getLogger("meatheadgear.catalog")

    async def sync_catalog():
        """Fetch products from Printful, calculate retail prices, upsert to SQLite."""
        if not settings.printful_api_key:
            logger.warning("PRINTFUL_API_KEY not set — skipping catalog sync")
            return

        client = PrintfulClient(settings.printful_api_key)
        products = await client.get_catalog_products()
        logger.info("Fetched %d gym wear products from Printful", len(products))

        async with aiosqlite.connect(str(DB_PATH)) as db:
            for product in products:
                # Upsert product
                await db.execute(
                    """INSERT INTO products (printful_id, name, description, category, thumbnail_url, synced_at)
                       VALUES (?, ?, ?, ?, ?, ?)
                       ON CONFLICT(printful_id) DO UPDATE SET
                           name=excluded.name, description=excluded.description,
                           category=excluded.category, thumbnail_url=excluded.thumbnail_url,
                           synced_at=excluded.synced_at""",
                    (product["id"], product.get("title", ""), product.get("description", ""),
                     product.get("type_name", ""), product.get("image", ""),
                     datetime.now(timezone.utc).isoformat())
                )

                # Fetch detailed product info for variants
                details = await client.get_product_details(product["id"])
                if not details:
                    continue

                # Get the local product ID
                cursor = await db.execute(
                    "SELECT id FROM products WHERE printful_id = ?", (product["id"],)
                )
                row = await cursor.fetchone()
                if not row:
                    continue
                local_product_id = row[0]

                # Upsert variants
                for variant in details.get("variants", []):
                    printful_price = float(variant.get("price", 0))
                    retail_price = calculate_retail_price(printful_price, settings.target_margin)
                    await db.execute(
                        """INSERT INTO product_variants
                           (product_id, printful_variant_id, name, size, color, color_hex,
                            printful_price, retail_price, in_stock, synced_at)
                           VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
                           ON CONFLICT(printful_variant_id) DO UPDATE SET
                               name=excluded.name, size=excluded.size, color=excluded.color,
                               color_hex=excluded.color_hex, printful_price=excluded.printful_price,
                               retail_price=excluded.retail_price, in_stock=excluded.in_stock,
                               synced_at=excluded.synced_at""",
                        (local_product_id, variant["id"], variant.get("name", ""),
                         variant.get("size", ""), variant.get("color", ""),
                         variant.get("color_code", ""), printful_price, retail_price,
                         1 if variant.get("availability_status") != "discontinued" else 0,
                         datetime.now(timezone.utc).isoformat())
                    )

                # Upsert images
                for idx, image in enumerate(details.get("images", [])):
                    await db.execute(
                        """INSERT INTO product_images (product_id, url, position)
                           VALUES (?, ?, ?)
                           ON CONFLICT DO NOTHING""",
                        (local_product_id, image.get("url", ""), idx)
                    )

            await db.commit()
        logger.info("Catalog sync complete — %d products synced", len(products))

    async def start_background_sync():
        """Start background task that re-syncs catalog every N hours."""
        interval = settings.printful_sync_interval_hours * 3600
        while True:
            await asyncio.sleep(interval)
            try:
                await sync_catalog()
            except Exception as e:
                logger.error("Background catalog sync failed: %s", e)
    ```

    **routers/catalog.py** — Product listing and detail endpoints:

    - `GET /products` — Query products table with optional `?category=` filter. For each product, compute `price_range` by querying MIN/MAX retail_price from product_variants. Return list of `{"id", "printful_id", "name", "category", "thumbnail_url", "price_min", "price_max", "is_active"}`.
    - `GET /products/{product_id}` — Query single product by local `id`. Include `variants` array (each with `id`, `name`, `size`, `color`, `color_hex`, `retail_price`, `in_stock`) and `images` array (each with `url`, `position`). Return 404 if product not found.

    ```python
    from fastapi import APIRouter, Depends, HTTPException
    from database import get_db

    router = APIRouter()

    @router.get("/products")
    async def list_products(category: str | None = None, db=Depends(get_db)):
        query = "SELECT id, printful_id, name, category, thumbnail_url, is_active FROM products WHERE is_active = 1"
        params = []
        if category:
            query += " AND LOWER(category) LIKE ?"
            params.append(f"%{category.lower()}%")

        cursor = await db.execute(query, params)
        products = await cursor.fetchall()

        result = []
        for p in products:
            price_cursor = await db.execute(
                "SELECT MIN(retail_price), MAX(retail_price) FROM product_variants WHERE product_id = ? AND in_stock = 1",
                (p[0],)
            )
            prices = await price_cursor.fetchone()
            result.append({
                "id": p[0], "printful_id": p[1], "name": p[2],
                "category": p[3], "thumbnail_url": p[4],
                "price_min": prices[0] or 0, "price_max": prices[1] or 0,
            })
        return {"products": result}

    @router.get("/products/{product_id}")
    async def get_product(product_id: int, db=Depends(get_db)):
        cursor = await db.execute(
            "SELECT id, printful_id, name, description, category, thumbnail_url FROM products WHERE id = ? AND is_active = 1",
            (product_id,)
        )
        product = await cursor.fetchone()
        if not product:
            raise HTTPException(status_code=404, detail="Product not found")

        var_cursor = await db.execute(
            "SELECT id, name, size, color, color_hex, retail_price, in_stock FROM product_variants WHERE product_id = ?",
            (product_id,)
        )
        variants = [
            {"id": v[0], "name": v[1], "size": v[2], "color": v[3],
             "color_hex": v[4], "retail_price": v[5], "in_stock": bool(v[6])}
            for v in await var_cursor.fetchall()
        ]

        img_cursor = await db.execute(
            "SELECT url, position FROM product_images WHERE product_id = ? ORDER BY position",
            (product_id,)
        )
        images = [{"url": i[0], "position": i[1]} for i in await img_cursor.fetchall()]

        return {
            "id": product[0], "printful_id": product[1], "name": product[2],
            "description": product[3], "category": product[4], "thumbnail_url": product[5],
            "variants": variants, "images": images,
        }
    ```
