    ```python
    _CLI_SETUP_STATE_FILE = Path(__file__).parent.parent / ".frood" / "cli-setup-state.json"


    def _load_cli_setup_state() -> dict:
        """Load persisted enabled/disabled state for cross-CLI setup."""
        import json
        if _CLI_SETUP_STATE_FILE.exists():
            try:
                return json.loads(_CLI_SETUP_STATE_FILE.read_text())
            except Exception:
                pass
        return {"enabled_clis": []}


    def _save_cli_setup_state(enabled_clis: list[str]) -> None:
        """Persist CLI setup enabled list to disk."""
        import json
        _CLI_SETUP_STATE_FILE.parent.mkdir(parents=True, exist_ok=True)
        _CLI_SETUP_STATE_FILE.write_text(
            json.dumps({"enabled_clis": sorted(set(enabled_clis))}, indent=2)
        )
    ```

    STEP B — Add the two endpoints inside `create_app`, near the existing tools endpoint block. Put
    them AFTER the `/api/tools/{name}` PATCH endpoint (around line 2519) so related admin endpoints

    ```python
    # -- CLI Setup (Phase 01: cross-cli-setup-core) ---------------------------

    @app.get("/api/cli-setup/detect")
    async def cli_setup_detect(_admin: AuthContext = Depends(require_admin)):
        """Report installed CLIs + current wiring state (DASH-01)."""
        from core.cli_setup import detect_all
        try:
            return detect_all()
        except Exception as e:
            raise HTTPException(status_code=500, detail=f"detect failed: {e}")

    class CliSetupWireRequest(BaseModel):
        cli: str
        enabled: bool

    @app.post("/api/cli-setup/wire")
    async def cli_setup_wire(
        req: CliSetupWireRequest,
        _admin: AuthContext = Depends(require_admin),
    ):
        """Wire or unwire the named CLI (DASH-02)."""
        from core.cli_setup import wire_cli, unwire_cli
        if req.cli not in ("claude-code", "opencode"):
            raise HTTPException(status_code=400, detail=f"unknown cli: {req.cli}")
        try:
            if req.enabled:
                result = wire_cli(req.cli)
                action = "wire"
            else:
                result = unwire_cli(req.cli)
                action = "unwire"
        except Exception as e:
            raise HTTPException(status_code=500, detail=f"{('wire' if req.enabled else 'unwire')} failed: {e}")

        # Persist toggle state (DASH-04)
        state = _load_cli_setup_state()
        enabled = set(state.get("enabled_clis", []))
        if req.enabled:
            enabled.add(req.cli)
        else:
            enabled.discard(req.cli)
        _save_cli_setup_state(sorted(enabled))

        return {"cli": req.cli, "enabled": req.enabled, "action": action, "result": result}
    ```

    ```javascript
    // ---- CLI Setup panel (Phase 01: cross-cli-setup-core, DASH-03) ----
    async function renderCliSetupPanel(container) {
      container.innerHTML = '<div class="loading">Detecting CLIs...</div>';
      let detected;
      try {
        const resp = await fetch('/api/cli-setup/detect', { credentials: 'include' });
        if (!resp.ok) throw new Error(`HTTP ${resp.status}`);
        detected = await resp.json();
      } catch (e) {
        container.innerHTML = `<div class="error">Failed to detect CLIs: ${e.message}</div>`;
        return;
      }

      const rows = Object.entries(detected).map(([cli, info]) => {
        const installed = info.installed ? 'installed' : 'not installed';
        const wired = info.wired ? 'wired' : 'not wired';
        return `
          <div class="cli-setup-row" data-cli="${cli}">
            <div class="cli-setup-label">
              <strong>${cli}</strong>
              <span class="muted">${installed} · ${wired}</span>
            </div>
            <label class="switch">
              <input type="checkbox" data-cli="${cli}" ${info.wired ? 'checked' : ''}>
              <span class="slider"></span>
            </label>
          </div>
        `;
      }).join('');

      container.innerHTML = `
        <div class="cli-setup-panel">
          <h2>CLI Setup</h2>
          <p class="blurb">
            Frood can wire itself into other MCP-capable CLIs so they can call
            <code>frood_skill</code> to load warehoused skills/commands/agents on demand.
            <a href="https://github.com/anthropics/claude-code" target="_blank" rel="noopener">Docs</a>
          </p>
          <div class="cli-setup-rows">${rows}</div>
        </div>
      `;

      container.querySelectorAll('input[type=checkbox][data-cli]').forEach(el => {
        el.addEventListener('change', async (ev) => {
          const cli = ev.target.dataset.cli;
          const enabled = ev.target.checked;
          ev.target.disabled = true;
          try {
            const resp = await fetch('/api/cli-setup/wire', {
              method: 'POST',
              headers: { 'Content-Type': 'application/json' },
              credentials: 'include',
              body: JSON.stringify({ cli, enabled })
            });
            if (!resp.ok) {
              const err = await resp.text();
              alert(`Failed: ${err}`);
              ev.target.checked = !enabled;
            }
          } catch (e) {
            alert(`Request failed: ${e.message}`);
            ev.target.checked = !enabled;
          } finally {
            ev.target.disabled = false;
            // Re-render to reflect latest detection state
            renderCliSetupPanel(container);
          }
        });
      });
    }
    ```
